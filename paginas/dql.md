## SELECT

### Sintaxe

```sql
SELECT column1, column2, ...
FROM table_name;

SELECT * FROM table_name;

SELECT column1, column2, ...
FROM table_name
WHERE condition;
```

### Exemplos
```sql
/*INNER JOIN*/
SELECT A.NOME_PROFESSOR AS PROFESSORES, B.NOME_MATERIA AS MATERIAS 
FROM professor A JOIN MATERIA B ON B.id_professor = A.id_professor WHERE nome_professor = 'LUIX'


SELECT A.NOME_ALUNO AS ALUNOS, B.NOME_MATERIA AS MATERIAS, C.NOTA AS NOTAS, D.NOME_PROFESSOR
FROM ALUNO A INNER JOIN MATERIA B ON B.ID_MATERIA = A.ID_ALUNO
INNER JOIN NOTA C ON C.ID_MATERIA = B.ID_MATERIA
INNER JOIN PROFESSOR D ON D.ID_PROFESSOR = B.ID_PROFESSOR
```

### View
```sql
/*CRIANDO UMA VIEW*/
CREATE OR ALTER VIEW NOTAS_ESCOLA AS
SELECT A.NOME_ALUNO AS ALUNOS, B.NOME_MATERIA AS MATERIAS, C.NOTA AS NOTAS, D.NOME_PROFESSOR
FROM ALUNO A INNER JOIN MATERIA B ON B.ID_MATERIA = A.ID_ALUNO
INNER JOIN NOTA C ON C.ID_MATERIA = B.ID_MATERIA
INNER JOIN PROFESSOR D ON D.ID_PROFESSOR = B.ID_PROFESSOR
```

### Funções de agrupamento
```sql
--COUNT -- CONTAGEM
SELECT COUNT(NOTAS) AS NOTAS1
FROM NOTAS_ESCOLA
WHERE ALUNOS = 'LUCAS'

--AVG -- MÉDIA
SELECT AVG(NOTAS) AS NOTAS1
FROM NOTAS_ESCOLA

--SUM -- SOMA
SELECT SUM(NOTAS) AS NOTAS1
FROM NOTAS_ESCOLA

--MIN -- MENOR VALOR(MINIMO)
SELECT MIN(NOTAS) AS NOTAS1
FROM NOTAS_ESCOLA

--MAX -- MAIOR VALOR(MAXIMO)
SELECT MAX(NOTAS) AS NOTAS1
FROM NOTAS_ESCOLA
```
